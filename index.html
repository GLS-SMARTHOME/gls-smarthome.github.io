function playCaptureAnimation(callback) { const flash = document.getElementById("captureFlash"); const anim = document.getElementById("captureAnimation"); // Start flash flash.classList.add("flashActive"); // Show pokeball anim.style.display = "flex"; // End animation after 1.5s setTimeout(() => { anim.style.display = "none"; flash.classList.remove("flashActive"); if (callback) callback(); }, 1500); }
